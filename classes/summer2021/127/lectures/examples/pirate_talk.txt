


def translator(sentence, pirate):

    word_list = sentence.split(" ")

    translated_words = []
    for each_word in word_list:
        
        t = pirate.get(each_word, each_word)

        translated_words.append(t)
        

    pirate_sentence = " ".join(translated_words)

    return pirate_sentence
