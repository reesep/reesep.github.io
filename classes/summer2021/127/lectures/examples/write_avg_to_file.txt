

file = open("test_scores.txt","r")

output_file = open("out.txt","w")

# no column header --> no need for .readline()

for each_line in file:
    row = each_line.strip().split(" ")

    
    scores = row[1:]
    for x in range(len(scores)):
        scores[x] = int(scores[x])

    average = sum(scores) / len(scores)
    name = row[0]
    
    output_file.write(name + " had an average score of " + str(average)+ "\n")



output_file.close()
file.close()
