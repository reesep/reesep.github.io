
# Write a python function that will take in a list of
# test scores (of any size) and returns the average of
# those exam scores




def average_score(exam_scores):
    size = len(exam_scores)
    s = 0
    for each_num in exam_scores:
        s += each_num
    
    answer = s / size
    return(answer)




scores = [100,90,80]
avg = average_score(scores)
print("Average Score:",avg)
