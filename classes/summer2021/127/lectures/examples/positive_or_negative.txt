

# print out whether a number is positive or negative



def positive_or_negative(num):

    if(num > 0):
        return "positive"
    elif(num < 0):
        return "negative"
    else:
        return "neither"


def main():
    
    num = int(input("Enter a number: "))
    answer = positive_or_negative(num)
    print(answer)

    
main()
