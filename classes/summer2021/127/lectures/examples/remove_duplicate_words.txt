

# Write a python function that will take in a list of
# words and return a list with all duplicate words
# removed


# input -> ["Hello","Dog","Hello","World"]
# output -> ["Hello", "Dog", "World"]


def remove_duplicate_words(list_of_words):

    no_duplicates = []
    for each_word in list_of_words:
        
        if each_word not in no_duplicates:
            no_duplicates.append(each_word)

    return no_duplicates



def main():

    words = ["Hello","Dog"]
    no_duplicate = remove_duplicate_words(words)
    print(no_duplicate)

main()
