## Bar graph of top 5 strongest water type pokemon

#create new dataframe with only water type pokemon
water_pokemon = df.loc[(df['type1'] == "Water") | (df['type2'] == "Water")]

#sort by most highest stat totals
water_pokemon = water_pokemon.sort_values('total',ascending=False)

#take a slice of top 5 dataframe rows
top_five = water_pokemon[0:5]

print(top_five)


top_five.plot.bar(x="name", y='total',rot=0,legend=None,color=["blue","lightblue"],edgecolor="black")


plt.xlabel("Name")
plt.ylabel("Stat Total")
plt.title("Top 5 Strongest Water Type Pokemon")

plt.show()
