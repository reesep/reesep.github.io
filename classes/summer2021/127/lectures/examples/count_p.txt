
# Write a python function count_p() that will take in
# a word. The function will count how many Pâ€™s are in
# that word


def count_p(word):
    count = 0
    for each_letter in word:

        if(each_letter == "p"):
            count += 1
    return count




def main():
    print(count_p("pepper"))
    print(count_p("pie"))
    print(count_p("apple"))
    print(count_p("reese"))
    print(count_p(""))

main()
