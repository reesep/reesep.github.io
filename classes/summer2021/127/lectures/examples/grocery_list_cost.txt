
#write a program that will sum of the total cost
# of a grocery list (grocery_list.txt)



file = open("grocery_list.txt","r")
total = 0

file.readline() #remove column headers

for each_line in file:
    each_line = each_line.strip() #remove all \n
    row = each_line.split(" ")

    
    cost = int(row[1]) * float(row[2])

    total += cost
    

print(total)

file.close()
